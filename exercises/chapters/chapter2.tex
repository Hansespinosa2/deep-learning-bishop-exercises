\begin{gather}
  P[C=1|T=1] = \frac{P[T=1|C=1]*P[C=1]}{P[T=1|C=1]*P[C=1] + P[T=1|C=0]*P[C=0]} \\
  P[C=1|T=1]= \frac{0.90*0.001}{0.90*0.001 + 0.03*0.999} = 0.0292
\end{gather}
